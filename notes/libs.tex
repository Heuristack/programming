int main()
{
  map<char,string> m = {{'H',"Ocean Park!"}};
  if (auto [i,r] = m.insert({'H',"Disneyland!"}); !r) {
    cout << get<0>(*i) << ": " << get<string>(*i) << endl;
  }
}

\end{code}
